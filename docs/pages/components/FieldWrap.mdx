import { FieldWrap, TextField, Button } from '@deque/cauldron-react'

```js
import { FieldWrap } from '@deque/cauldron-react'
```

## Examples

### Single Field

```jsx example
<FieldWrap>
  <TextField label="First name" />
</FieldWrap>
```

### Multiple Fields

```jsx example
<FieldWrap as="fieldset">
  <legend>
    Send us a message
  </legend>
  <TextField label="Email" />
  <TextField
    label="Message"
    multiline
  />
  <Button>
    Send Message
  </Button>
</FieldWrap>
```
