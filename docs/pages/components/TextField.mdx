import { TextField, FieldWrap } from '@deque/cauldron-react'

```js
import { TextField } from '@deque/cauldron-react'
```

<Note>
  Use [FieldWrap](/FieldWrap) to wrap all input components. The `TextField` component has been designed specifically to only be used when rendered as a child of this component, and may not be completely accessible if rendered outside of `FieldWrap`.
</Note>

## Examples

### Resting

```jsx example
<FieldWrap>
  <TextField label="Resting" />
</FieldWrap>
```

### Disabled

```jsx example
<FieldWrap>
  <TextField
    label="Disabled"
    disabled
  />
</FieldWrap>
```

### Required

```jsx example
<FieldWrap>
  <TextField
    label="Required"
    required
  />
</FieldWrap>
```

### Error

```jsx example
<FieldWrap>
  <TextField
    label="Error"
    error="This field has an error!"
  />
</FieldWrap>
```
