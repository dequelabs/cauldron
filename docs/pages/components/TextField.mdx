import { TextField, FieldWrap } from '@deque/cauldron-react';

```js
import { TextField } from '@deque/cauldron-react';
```

<Note>
  Use [FieldWrap](./FieldWrap) to wrap all input components. The `TextField`
  component has been designed specifically to only be used when rendered as a
  child of this component, and may not be completely accessible if rendered
  outside of `FieldWrap`.
</Note>

## Examples

### Resting

```jsx example
<FieldWrap>
  <TextField label="Resting" />
</FieldWrap>
```

### Disabled

```jsx example
<FieldWrap>
  <TextField label="Disabled" disabled />
</FieldWrap>
```

### Required

```jsx example
<FieldWrap>
  <TextField label="Required" required />
</FieldWrap>
```

### Error

```jsx example
<FieldWrap>
  <TextField label="Error" error="This field has an error!" required />
</FieldWrap>
```

### With Description

```jsx example
<FieldWrap>
  <TextField label="Description" description="This field has a description" />
</FieldWrap>
```

### Multi-line Resting

```jsx example
<FieldWrap>
  <TextField label="Multiline" multiline />
</FieldWrap>
```

### Multi-line Disabled

```jsx example
<FieldWrap>
  <TextField label="Multiline Disabled" multiline disabled />
</FieldWrap>
```

### Multi-line Required

```jsx example
<FieldWrap>
  <TextField label="Multiline Required" multiline required />
</FieldWrap>
```

### Multi-line Error

```jsx example
<FieldWrap>
  <TextField
    label="Multiline Error"
    multiline
    error="This field has an error!"
  />
</FieldWrap>
```
