import { OptionsMenu, OptionsMenuItem, OptionsMenuTrigger, Icon, Button } from '@deque/cauldron-react'

```jsx
import { 
  OptionsMenu, 
  OptionsMenuItem, 
  OptionsMenuTrigger
} from '@deque/cauldron-react'
```

## Examples

### Default Trigger

```jsx example
<OptionsMenu
  align="left"
  trigger={triggerProps => (
    <OptionsMenuTrigger {...triggerProps}>
      <Icon type="kabob" label="Options" />
    </OptionsMenuTrigger>
  )}
>
  <OptionsMenuItem>This</OptionsMenuItem>
  <OptionsMenuItem>That</OptionsMenuItem>
  <OptionsMenuItem>The third</OptionsMenuItem>
  <OptionsMenuItem disabled>And the other</OptionsMenuItem>
</OptionsMenu>
```

### Alternate Trigger

The trigger can be any type of component that acts as a button, allowing for advanced customization of the trigger:

```jsx example
<OptionsMenu
  align="left"
  trigger={triggerProps => (
    <Button variant="secondary" {...triggerProps}>
      Options
      <Icon type="arrow-circle-down" />
    </Button>
  )}
>
  <OptionsMenuItem>This</OptionsMenuItem>
  <OptionsMenuItem>That</OptionsMenuItem>
  <OptionsMenuItem>The third</OptionsMenuItem>
  <OptionsMenuItem disabled>And the other</OptionsMenuItem>
</OptionsMenu>
```
