import { Card, CardHeader, CardContent, CardFooter, Toast } from '@deque/cauldron-react'

```js
import {
  Card,
  CardHeader,
  CardContent,
  CardFooter
} from '@deque/cauldron-react'
```

## Examples

```jsx example
<Card>
  <CardHeader>
    <h3>Card heading</h3>
  </CardHeader>
  <CardContent>
    <p>Card content</p>
  </CardContent>
  <CardFooter>Footer content</CardFooter>
</Card>
````

### Simple

```jsx example
<Card variant="simple">
  <CardHeader>
    <h3>Card heading</h3>
  </CardHeader>
  <CardContent>
    <p>Card content</p>
  </CardContent>
</Card>
```
