function TreeViewExample() {
  const onAction = (key) => {
    const elm = document.querySelector(`div[data-key="${key}"]`);
    const selected = elm.getAttribute('aria-selected', 'true');
    elm.setAttribute('aria-selected', selected === 'true' ? 'false' : 'true');
  };
  const items = [
    {
      id: '1',
      textValue: 'Documents',
      children: [
        {
          id: '2',
          textValue: 'Project',
          children: [
            { id: '3', textValue: 'Weekly Report' },
            { id: '4', textValue: 'Budget' }
          ]
        }
      ]
    },
    {
      id: '5',
      textValue: 'Photos',
      children: [
        { id: '6', textValue: 'Image 1' },
        { id: '7', textValue: 'Image 2' }
      ]
    }
  ];
  return (
    <TreeView
      selectionMode="none"
      defaultExpandedKeys={['1', '2']}
      items={items}
    />
  );
}
