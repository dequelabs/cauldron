import { FieldGroup, TextField, RadioGroup, Checkbox } from '@deque/cauldron-react'

```js
import { FieldGroup } from '@deque/cauldron-react'
```

## Examples

### Grouping Checkboxes

```jsx example
<FieldGroup label="Favorite way to prepare water">
  <Checkbox 
    id="grilled-water"
    label="Grilled" 
    name="prepare-water" 
    value="grilled-water"
    checked
  />
  <Checkbox 
    id="toasted-water"
    label="Toasted" 
    name="prepare-water" 
    value="toasted-water"
  />
  <Checkbox 
    id="steamed-water"
    label="Steamed" 
    name="prepare-water" 
    value="steamed-water"
  />
  <Checkbox 
    id="fried-water"
    label="Fried" 
    name="prepare-water" 
    value="fried-water"
  />
</FieldGroup>
```

### Grouping Radio Buttons

```jsx example
<FieldGroup label="How big is this thing?">
  <RadioGroup
    name="size"
    radios={[
      {
        id: '1',
        label: '1',
        value: '1'
      },
      {
        id: '2',
        label: '2',
        value: '2'
      },
      {
        id: '3',
        label: '3',
        value: '3'
      }
    ]}
  />
</FieldGroup>
```

### With Description

```jsx example
<FieldGroup 
  label="Checkboxes" 
  description="Please check at least one checkbox"
>
  <Checkbox 
    id="yes"
    label="Yes" 
    name="checkboxes" 
  />
  <Checkbox 
    id="no"
    label="No" 
    name="checkboxes" 
  />
  <Checkbox 
    id="why"
    label="Why?" 
    name="checkboxes" 
  />
</FieldGroup>
```

### With Error Message

```jsx example
<FieldGroup 
  label="Please choose your favorite colors" 
  description="Your favorite color can not be blue"
  error="You cannot choose blue as one of your favorite colors"
>
  <Checkbox 
    id="red"
    label="Red" 
    name="colors" 
  />
  <Checkbox 
    id="blue"
    label="Blue" 
    name="colors"
    checked
  />
  <Checkbox 
    id="green"
    label="Green" 
    name="colors"
  />
</FieldGroup>
```

## Props

<ComponentProps
  children={true}
  className={true}
  refType="HTMLDivElement"
  props={[
    {
      name: 'label',
      type: 'React.ReactNode',
      required: true,
      description: 'Label text for the group of form elements.'
    },
    {
      name: 'description',
      type: 'React.ReactNode',
      description: 'Optional descriptive text.'
    },
    {
      name: 'error',
      type: 'React.ReactNode',
      description: 'Error message to display when validation fails for any of the form elements in the group.'
    }
  ]}
/>
