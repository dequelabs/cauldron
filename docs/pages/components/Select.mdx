import { useState } from 'react';
import { FieldWrap, Select } from '@deque/cauldron-react';

```js
import { Select } from '@deque/cauldron-react';
```

<Note>
  Input components _should_ be wrapped in a [FieldWrap](./FieldWrap). The
  `Select` component has been designed specifically to meet color contrast
  requirements against a `FieldWrap` background.

  If you use a `Select` without a `FieldWrap` (for example, within a `Modal`,
  where `FieldWrap`'s `Panel` appearance might not be appropriate), you _must_
  verify that it meets accessibility requirements in all component states,
  especially for color contrast.
</Note>

### Uncontrolled

Select can be uncontrolled or controlled. For an uncontrolled component, use `defaultValue` to set the initial value of the listed options.

```jsx example
<FieldWrap>
  <Select
    label="Uncontrolled Select"
    options={[
      {
        key: 'yes',
        value: 'Yes'
      },
      {
        key: 'no',
        value: 'No'
      },
      {
        key: 'maybe',
        value: 'Maybe'
      }
    ]}
    defaultValue="Yes"
  />
</FieldWrap>
```

### Controlled

For a controlled select, use `value` to set the currently selected value of the listed options. You must also use this in conjunction with `onChange` for the value to get updated on change.

```jsx example
function ControlledSelectExample() {
  const [value, setValue] = useState('Yes');
  const handleChange = (event) => {
    setValue(event.target.value);
  };
  return (
    <FieldWrap>
      <Select
        label="Controlled Select"
        options={[
          {
            key: 'yes',
            value: 'Yes'
          },
          {
            key: 'no',
            value: 'No'
          },
          {
            key: 'maybe',
            value: 'Maybe'
          }
        ]}
        value={value}
        onChange={handleChange}
      />
    </FieldWrap>
  );
}
```

### Options

The options list can be an array of objects with specific properties, or an `HTMLOptions` element:

```jsx example
function OptionsExample() {
  const options = ['one', 'two', 'three'];
  return (
    <FieldWrap>
      <Select
        label="Select with array"
        options={options.map((value) => ({ key: value, value }))}
      />
      <Select label="Select with HTMLOptions">
        {options.map((value) => (
          <option key={value} value={value}>
            {value}
          </option>
        ))}
      </Select>
    </FieldWrap>
  );
}
```
