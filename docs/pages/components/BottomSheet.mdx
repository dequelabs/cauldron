## Examples

```jsx example
function BottomSheetExample() {
  const [open, setOpen] = useState(false);
  const toggleOpen = () => setOpen(!open);
  return (
    <>
      <IconButton label="Toggle Bottom Sheet" icon="hamburger-menu" onClick={toggleOpen} />
      <BottomSheet label="Bottom Sheet Title" open={open} onClose={toggleOpen}>
        Contents
      </BottomSheet>
    </>
  )
}
```
