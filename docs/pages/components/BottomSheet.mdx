import { useState } from 'react';
import { BottomSheet, IconButton, Icon, Button } from '@deque/cauldron-react';

```js
import { BottomSheet } from '@deque/cauldron-react';
```

`BottomSheet` is an extension of the [`Drawer`](./Drawer) component to display additional information, typically on smaller devices.

## Examples

```jsx example
function BottomSheetExample() {
  const [open, setOpen] = useState(false);
  const toggleOpen = () => setOpen(!open);
  return (
    <>
      <IconButton
        label="Toggle Bottom Sheet"
        icon="hamburger-menu"
        onClick={toggleOpen}
      />
      <BottomSheet label="Bottom Sheet Title" open={open} onClose={toggleOpen}>
        Contents
      </BottomSheet>
    </>
  );
}
```
