### Selections (Single)

```jsx example
function SingleSelectExample() {
  const options = ['One', 'Two', 'Three'];
  const [selected, setSelected] = useState('One');
  const handleAction = (key) => {
    setSelected(key)
  };

  return (
    <ActionMenu trigger={<Button variant="secondary">Click me!</Button>}>
      <ActionList selectionType="single">
        {options.map(value => (
          <ActionListItem 
            key={value} 
            selected={selected === value}
            onAction={() => handleAction(value)}
          >
            {value}
          </ActionListItem>
        ))}
      </ActionList>
    </ActionMenu>
  )
}
```

```jsx example
function MultiSelectExample() {
  const options = ['One', 'Two', 'Three'];
  const [selected, setSelected] = useState([]);
  const handleAction = (key) => {
    if (selected.includes(key)) {
      setSelected(selected.filter(v => v !== key ));
    } else {
      setSelected([...selected, key]);
    }
  };

  return (
    <ActionMenu trigger={<Button variant="secondary">Click me!</Button>}>
      <ActionList selectionType="single">
        {options.map(value => (
          <ActionListItem 
            key={value} 
            selected={selected.includes(value)}
            onAction={() => handleAction(value)}
          >
            {value}
          </ActionListItem>
        ))}
      </ActionList>
    </ActionMenu>
  )
}
```
